package com.example.examplexml

import android.os.Bundle
import android.widget.Button
import android.widget.TextView
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    // --- SECCIÓN ESTRICTAMENTE DE FUNCIONALIDAD ---
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        // Le decimos a este archivo que su diseño visual es el archivo XML
        setContentView(R.layout.activity_main)

        // Atrapamos los elementos visuales usando los IDs que pusimos en el XML
        val textoMensaje = findViewById<TextView>(R.id.textoMensaje)
        val botonIniciar = findViewById<Button>(R.id.botonIniciar)

        // Le programamos qué hacer al detectar un clic
        botonIniciar.setOnClickListener {
            textoMensaje.text = "Cargando... "
        }
    }
}
